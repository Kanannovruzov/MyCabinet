        }

            if(item.status==='Müddəti bitib') return 'Müddəti bitib';

    (function(){
        const MIN_SECONDS = 30 * 60;
        const PING_EVERY  = 25;

        let pingTimer = null;
        let isReadingActive = false;

        function setReadingActive(v){
            isReadingActive = !!v;
        }

        function isHeyzineModalOpen(){
            return !!document.querySelector(
                '.heyzine-modal, .heyzinea-modal, .hz-modal, .fp-modal, .fp-overlay, .modal.heyzine'
            );
        }

        const hzObserver = new MutationObserver(() => {
            setReadingActive(isHeyzineModalOpen());
        });

        function startObserveHeyzine(){
            try{
                hzObserver.observe(document.body, { childList:true, subtree:true });
                setReadingActive(isHeyzineModalOpen());
            }catch(e){}
        }
        function stopObserveHeyzine(){
            try{ hzObserver.disconnect(); }catch(e){}
            setReadingActive(false);
        }

        let statusTimer = null;

        function setBtnState(trainingId, enabled, text){
            const btn = document.querySelector('.js-cert-btn[data-training-id="'+trainingId+'"]');
            const st  = document.querySelector('.js-training-status');
            if(btn){
                btn.disabled = !enabled;
                btn.style.opacity = enabled ? '1' : '.6';
                btn.style.cursor  = enabled ? 'pointer' : 'not-allowed';
            }
            if(st && text) st.textContent = text;
        }

        async function postJSON(url, data){
            const r = await fetch(url, {
                method: 'POST',
                headers: {'Content-Type':'application/json','X-Requested-With':'XMLHttpRequest'},
                body: JSON.stringify(data||{})
            });
            return await r.json();
        }

        async function getJSON(url){
            const r = await fetch(url, { headers: {'X-Requested-With':'XMLHttpRequest'}});
            return await r.json();
        }

        async function startTraining(trainingId){
            const res = await postJSON('<?= site_url('trainings/start') ?>', { training_id: trainingId });
            if(!res || !res.ok) throw new Error(res?.msg || 'start error');
            return res;
        }

        async function pingTraining(token){
            return await postJSON('<?= site_url('trainings/ping') ?>', { token: token });
        }

        async function statusTraining(trainingId){
            return await getJSON('<?= site_url('trainings/status') ?>?training_id=' + encodeURIComponent(trainingId));
        }

        function stopTimers(){
            if(pingTimer){ clearInterval(pingTimer); pingTimer=null; }
            if(statusTimer){ clearInterval(statusTimer); statusTimer=null; }
        }

        async function beginFlow(trainingId){
            stopTimers();

            const st = document.querySelector('.js-training-status');
            if(st) st.textContent = 'Yoxlanılır...';

            const started = await startTraining(trainingId);

            if(started.completed){
                setBtnState(trainingId, true, 'Oxunuş tamamlandı ✅');
                return;
            } else {
                setBtnState(trainingId, false, 'Oxunur... 15 dəq tələb olunur');

                if(!isReadingActive) return;
                try{
                    const pr = await pingTraining(started.token);

                    if(pr && pr.ok){
                        if(pr.completed){
                            setBtnState(trainingId, true, 'Oxunuş tamamlandı ✅');
                            stopTimers();
                        } else {
                            const remain = pr.remain_seconds ?? null;
                            if(remain != null){
                                const mm = String(Math.floor(remain/60)).padStart(2,'0');
                                const ss = String(remain%60).padStart(2,'0');
                                setBtnState(trainingId, false, `Oxunur... qalıb ${mm}:${ss}`);
                            }
                        }
                    }
                }catch(e){}
            }, PING_EVERY * 1000);

            statusTimer = setInterval(async () => {
                if(!isReadingActive) return;
                try{
                    const sr = await statusTraining(trainingId);
                    if(sr && sr.ok && sr.completed){
                        setBtnState(trainingId, true, 'Oxunuş tamamlandı ✅');
                        stopTimers();
                    }
                }catch(e){}
            }, 5000);

    (async function(){
        const audio = document.getElementById('notifAudio');
        const SOUND_KEY = 'notif_sound_allowed';
        let PENDING_BEEP = false;

        if (!audio) return;
        audio.volume = 1;
        try { audio.load(); } catch(e) {}

        function soundAllowed(){ return localStorage.getItem(SOUND_KEY) === '1'; }
        function setSoundAllowed(v){ localStorage.setItem(SOUND_KEY, v ? '1' : '0'); }

        async function unlockSound(){
            try{
                audio.currentTime = 0;
                await audio.play();
                audio.pause();
                audio.currentTime = 0;
                setSoundAllowed(true);
                return true;
            }catch(e){
                return false;
            }
        }

        async function playBeep(){
            try{
                audio.currentTime = 0;
                await audio.play();
                PENDING_BEEP = false;
                return true;
            }catch(e){
                PENDING_BEEP = true;
                return false;
            }
        }

        function hookInteractions(){
            const fire = async ()=>{
                if(!soundAllowed()) return;
                if(!PENDING_BEEP) return;

                const ok = await playBeep();
                if(ok) return;

                if (window.Swal && !Swal.isVisible()){
                    const r = await Swal.fire({
                        icon: 'info',
                        title: 'Səs aktiv deyil',
                        text: 'Bildiriş səsini aktiv etmək üçün “Aktiv et” basın.',
                        confirmButtonText: 'Aktiv et',
                        showCancelButton: true,
                        cancelButtonText: 'Sonra'
                    });
                    if(r.isConfirmed){
                        const u = await unlockSound();
                        if(u){
                            await playBeep();
                        }
                    }
                }
            };

            ['click','touchstart','keydown'].forEach(ev=>{
                window.addEventListener(ev, fire, { passive:true });
            });
        }
        hookInteractions();

        async function askSoundPermissionIfNeeded(){
            if(soundAllowed()) return;

            const r = await Swal.fire({
                icon: 'question',
                title: 'Bildiriş səsi aktiv edilsin?',
                text: 'Yeni bildirişlər üçün səsli xəbərdarlıq edək?',
                confirmButtonText: 'Bəli',
                cancelButtonText: 'Yox',
                showCancelButton: true,
                allowOutsideClick: false
            });

            if(!r.isConfirmed){
                setSoundAllowed(false);
                return;
            }

            const ok = await unlockSound();
            if(ok){
                await Swal.fire({ icon:'success', title:'Səs aktiv edildi', timer:900, showConfirmButton:false });
            }else{
               setSoundAllowed(true);
                PENDING_BEEP = true;
            }
        }


        async function checkNotificationsAndNotify(){
            try{
                const res = await fetch("<?= site_url('api/notifications') ?>", {
                    headers:{'X-Requested-With':'XMLHttpRequest'}
                });
                const data = await res.json();
                if(!data?.ok) return;

                const unread = Number(data.unread || 0);
                if(unread <= 0) return;

                    title:'Yeni bildirişiniz var',
                    text:`Oxunmamış bildiriş sayı: ${unread}`,
                    confirmButtonText:'Bax',
                    cancelButtonText:'Sonra',
                    showCancelButton:true,
                    allowOutsideClick:false
                });

                    await playBeep();
                }

                if(r.isConfirmed){
                    document.getElementById('notificationBtn')?.click();
                }
            }catch(e){
                console.warn(e);
            }
        }

        });

    (function(){
        const $m = $('#fbModal');
        const $msg = $('#fbMessage');
        const $send = $('#fbSend');
        // ===== Message limit settings =====
        const MAX_LEN = 2000;        // mesaj max simvol
        const MAX_LINES = 6;        // max sətir (enter) sayı

        // HTML tərəfdə maxlength yoxdursa, burda qoyaq
        try { $msg.attr('maxlength', MAX_LEN); } catch(e){}

        // counter UI (yoxdursa, yaradıb textarea altına qoyur)
        let $counter = $('#fbCounter');
        if(!$counter.length){
            $counter = $('<div id="fbCounter" style="margin-top:6px;color:#6b7280;font-size:12px;text-align:right;"></div>');
            $msg.after($counter);
        }

        function normalizeText(v){
            v = (v || '').replace(/\s{3,}/g,'  '); // çox boşluq sıxışdır
            // sətir limit
            const lines = v.split(/\r?\n/);
            if(lines.length > MAX_LINES){
                v = lines.slice(0, MAX_LINES).join('\n');
            }
            // uzunluq limit
            if(v.length > MAX_LEN){
                v = v.slice(0, MAX_LEN);
            }
            return v;
        }

        function updateCounter(){
            const val = ($msg.val() || '');
            const left = Math.max(0, MAX_LEN - val.length);
            $counter.text(left + ' simvol qalıb');
        }

        // typing
        $msg.on('input', function(){
            const fixed = normalizeText(this.value);
            if(fixed !== this.value) this.value = fixed;
            updateCounter();
        });

        // paste limit
        $msg.on('paste', function(){
            setTimeout(()=> {
                const fixed = normalizeText($msg.val());
                if(fixed !== $msg.val()) $msg.val(fixed);
                updateCounter();
            }, 0);
        });

        updateCounter();
        // ===== end =====

        function openFb(){
            $m.addClass('open').attr('aria-hidden','false');
            document.body.classList.add('body-lock');
            $msg.val('');
        }

                    Swal.fire({icon:'error', title:'Xəta', text: msg});
                })

                    $send.prop('disabled', false).css({opacity:1});
                });
        });
